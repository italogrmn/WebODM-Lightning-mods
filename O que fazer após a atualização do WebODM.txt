import shutil
import os
import time


def dispath():
    user_profile = os.environ['USERPROFILE']
    file_source = user_profile + r"\GRUPO EQUATORIAL ENERGIA\Projetos - varejo - Documentos\Projetos\Modelos e arquivos de configuracao\REVIT\Plugin Revit\Arquivos de Plugin"
    if os.path.exists(r'C:\\Plugin Revit'):
        shutil.rmtree(r'C:\\Plugin Revit')
    file_destination = 'C:\\Plugin Revit'
    shutil.copytree(file_source, file_destination)
    edit_dll(file_destination)
    if os.path.isfile(r'C:\ProgramData\Autodesk\Revit\Addins\2023\MeuPlugin.addin'):
        os.remove(r'C:\ProgramData\Autodesk\Revit\Addins\2023\MeuPlugin.addin')
    shutil.move(file_destination + '\\MeuPlugin.addin', r'C:\ProgramData\Autodesk\Revit\Addins\2023')


def edit_dll(local):
    caminho_arquivo = local + r'\MeuPlugin.addin'
    with open(caminho_arquivo, 'r') as arquivo:
        linhas = arquivo.readlines()
    nova_linha = '       <Assembly>'+ local + r'\Modelo2.dll</Assembly>\n'  # Nova linha desejada
    for i, linha in enumerate(linhas):
        if linha.strip().startswith("<Assembly>"):
            linhas[i] = nova_linha
            break
    with open(caminho_arquivo, 'w') as arquivo:
        arquivo.writelines(linhas)


try:

except Exception as e:
